int csv_set_field(CSV_BUFFER *buffer, size_t row, size_t entry,
                char *field)
{

        while (row >= buffer->rows) {
                append_row(buffer);
        }
        while (entry >= buffer->width[row])
                append_field(buffer, row);

        if (set_field(buffer->field[row][entry], field) == 0)
                return 0;
        else 
                return 1;
}
