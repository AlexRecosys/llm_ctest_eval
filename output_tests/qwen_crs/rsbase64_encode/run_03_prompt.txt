    rs32u_t len=src->n;

    size_t olen = len * 4 / 3 + 4; /* 3-byte blocks to 4-byte */

    if (ncol_max)
      olen += olen / ncol_max; /* line feeds */
    if (olen >= len) {
      rsbuffer_t *dest=rsbuffer_create((rs32u_t)olen,MA);
      if (dest) {
        rs8u_t *pos=dest->d;
        rs8u_t *end=src->d+len;
        rs8u_t *end3=end-3;
        rs8u_t *in=src->d;
        size_t line_len=0;

          if (ncol_max && line_len>=ncol_max) {
            *pos++ = '\n';
            line_len = 0;
          }

          in += 3;
          line_len += 4;
        }

          if (ncol_max && line_len >= ncol_max) {
            *pos++ = '\n';
            line_len = 0;
          }
          *pos++ = base64_table[in[0] >> 2];

            *pos++ = base64_table[(in[0] & 0x03) << 4];

          }
          *pos++ = '=';
        }
        dest->n=(rs32u_t)(pos-dest->d);
        return dest;
      }
    }
  } else if (src)
